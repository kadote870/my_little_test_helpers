# js-check-string-length

```js
const checkStringsLength = [
    'UPReQyTf9l2Ojfj',
    'wWOTafBp8Y8kkag',
    '6-Iu3s5rPwp-XON',
    'kWmuwpQuPDalxI7',
    'QJ5YTUbUpmg1N5Z',
    's9OrgOXk6Qco6pq',
    '1hXTfDTgFNzFfGM',
    '3s7XN4a5rDhO5Jg',
    '28v09gejFLD2wwE',
    'RTJd-Mra2JIU4UQ'
];


for (let i = 0; i < checkStringsLength.length; i++) {
    console.log(`Index no${checkStringsLength.indexOf(checkStringsLength[i])}; "${checkStringsLength[i]}"; length: ${checkStringsLength[i].length}`)
}
```
